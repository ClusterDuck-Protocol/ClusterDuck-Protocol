133template<typename T,size_t S>
134std::string arrayToHexString(const std::array<T,S> arr) {
135 std::string buf = ""; // static to avoid memory leak
136 buf.clear();
137 buf.reserve(S * 2); // 2 digit hex
138 const char* cs = "0123456789ABCDEF";
139 for (int i = 0; i < S; i++) {
140 uint8_t val = arr[i];
141 buf += cs[(val >> 4) & 0x0F];
142 buf += cs[val & 0x0F];
143 }
144 return buf;
145}
