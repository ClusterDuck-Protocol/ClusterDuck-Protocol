253 void attemptNetworkJoin(){
254 std::optional<CdpPacket> cdpNode = checkForNetworks();
255 if(cdpNode.has_value()){
256 //add an entry for the nearest neighbor, next hop is itself
257 router.insertIntoRoutingTable(cdpNode->sduid, cdpNode->sduid, this->getSignalScore()); //should signal score be stored on cdp packet?
258 router.setNetworkState(NetworkState::PUBLIC);
259 } else {
260 if((millis() - this->lastRreqTime) > NET_JOIN_DELAY){
261 sendRouteRequest(BROADCAST_DUID, getDuckId());
262 loginfo_ln("searching for networks....");
263 lastRreqTime = millis();
264 }

353 SignalScore getSignalScore(){
354 SignalScore signalInfo;
355 signalInfo.rssi = (duckRadio.getRSSI() - RSSI_MIN)/(RSSI_MAX-RSSI_MIN);
356 signalInfo.snr = (duckRadio.getSNR() - SNR_MIN)/(SNR_MAX-SNR_MIN);
357 signalInfo.signalScore = (signalInfo.rssi + signalInfo.snr) / 2.0f;
358 return signalInfo;
