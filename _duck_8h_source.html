284 void attemptNetworkJoin(){
285 std::optional<CdpPacket> cdpNode = checkForNetworks();
286 if(cdpNode.has_value()){
287 //add an entry for the nearest neighbor, next hop is itself
288 router.insertIntoRoutingTable(cdpNode->sduid, cdpNode->sduid, this->getSignalScore()); //should signal score be stored on cdp packet?
289 router.setNetworkState(NetworkState::PUBLIC);
290 } else {
291 if((millis() - this->lastRreqTime) > NET_JOIN_DELAY){
292 RouteJSON rreqDoc = RouteJSON(BROADCAST_DUID, this->duid);
293 rreqDoc.addToPath(this->duid);

388 SignalScore getSignalScore(){
389 SignalScore signalInfo;
390 signalInfo.rssi = (duckRadio.getRSSI() - RSSI_MIN)/(RSSI_MAX-RSSI_MIN);
391 signalInfo.snr = (duckRadio.getSNR() - SNR_MIN)/(SNR_MAX-SNR_MIN);
392 signalInfo.signalScore = (signalInfo.rssi + signalInfo.snr) / 2.0f;
393 return signalInfo;
